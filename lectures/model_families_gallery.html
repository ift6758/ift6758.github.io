def fit_and_plot(model, ads, eval_grid):
    model.fit(ads[["TV", "Radio"]], ads[["Sales"]])
    eval_grid["y_hat"] = model.predict(eval_grid[["tv", "radio"]])
    return sns.scatterplot(data=eval_grid, x="radio", y="y_hat", hue="tv")
